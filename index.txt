  <script>
    // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

    // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    let registeredUsers = JSON.parse(localStorage.getItem('registeredUsers')) || [];
    let currentUser = null;
    let isAdmin = false;

    let guildTestRegistrations = JSON.parse(localStorage.getItem('guildTestRegistrations')) || [];
    let guildVersusApplications = JSON.parse(localStorage.getItem('guildVersusApplications')) || [];
    let officerContacts = JSON.parse(localStorage.getItem('officerContacts')) || [];
    let matchRecords = JSON.parse(localStorage.getItem('matchRecords')) || [];
    let highlights = JSON.parse(localStorage.getItem('highlights')) || [];
    let merchandise = JSON.parse(localStorage.getItem('merchandise')) || [];
    let events = JSON.parse(localStorage.getItem('events')) || [];

    function saveAllData() {
      localStorage.setItem('registeredUsers', JSON.stringify(registeredUsers));
      localStorage.setItem('guildTestRegistrations', JSON.stringify(guildTestRegistrations));
      localStorage.setItem('guildVersusApplications', JSON.stringify(guildVersusApplications));
      localStorage.setItem('officerContacts', JSON.stringify(officerContacts));
      localStorage.setItem('matchRecords', JSON.stringify(matchRecords));
      localStorage.setItem('highlights', JSON.stringify(highlights));
      localStorage.setItem('merchandise', JSON.stringify(merchandise));
      localStorage.setItem('events', JSON.stringify(events));
    }

    // 90-day auto-login
    window.addEventListener('load', () => {
      const saved = JSON.parse(localStorage.getItem('currentUser'));
      if (saved && Date.now() - saved.timestamp < 90 * 24 * 60 * 60 * 1000) {
        currentUser = saved.user;
        isAdmin = saved.isAdmin;
        unlockSite();
        if (isAdmin) {
          document.getElementById('adminDashboard').classList.remove('hidden');
          switchAdminTab(0);
        }
      }
      renderAllClientPages();
    });

    function unlockSite() {
      document.getElementById('lockedScreen').classList.add('hidden');
      document.getElementById('mainContent').classList.remove('hidden');
      document.getElementById('navLinks').classList.remove('hidden');
      document.getElementById('authButtons').classList.add('hidden');
      document.getElementById('userArea').classList.remove('hidden');
      document.getElementById('usernameDisplay').textContent = currentUser ? currentUser.name : 'User';
    }

    function lockSite() {
      document.getElementById('lockedScreen').classList.remove('hidden');
      document.getElementById('mainContent').classList.add('hidden');
      document.getElementById('navLinks').classList.add('hidden');
      document.getElementById('authButtons').classList.remove('hidden');
      document.getElementById('userArea').classList.add('hidden');
      document.getElementById('adminDashboard').classList.add('hidden');
    }

    function showLoginModal() { document.getElementById('loginModal').classList.remove('hidden'); }
    function hideLoginModal() { document.getElementById('loginModal').classList.add('hidden'); }
    function showRegisterModal() { document.getElementById('registerModal').classList.remove('hidden'); }
    function hideRegisterModal() { document.getElementById('registerModal').classList.add('hidden'); }

    // Login / Register / Logout
    function handleLogin(e) {
      if (e) e.preventDefault();
      const email = document.getElementById('loginEmail').value.trim().toLowerCase();
      const pass = document.getElementById('loginPass').value.trim();
      const errorEl = document.getElementById('loginError');

      errorEl.classList.add('hidden');
      errorEl.textContent = '';

      if (email === "admin@haminepal.gg" && pass === "hamiadmin2026") {
        isAdmin = true;
        currentUser = { name: "Admin", email };
        localStorage.setItem('currentUser', JSON.stringify({ user: currentUser, isAdmin: true, timestamp: Date.now() }));
        hideLoginModal();
        unlockSite();
        document.getElementById('adminDashboard').classList.remove('hidden');
        switchAdminTab(0);
        return;
      }

      const user = registeredUsers.find(u => u.email.toLowerCase() === email && u.pass === pass);
      if (user) {
        currentUser = user;
        isAdmin = false;
        localStorage.setItem('currentUser', JSON.stringify({ user: currentUser, isAdmin: false, timestamp: Date.now() }));
        hideLoginModal();
        unlockSite();
      } else {
        errorEl.textContent = "Invalid credentials.";
        errorEl.classList.remove('hidden');
      }
    }

    function handleRegister(e) {
      if (e) e.preventDefault();
      const name = document.getElementById('regName').value.trim();
      const email = document.getElementById('regEmail').value.trim().toLowerCase();
      const pass = document.getElementById('regPass').value.trim();
      const errorEl = document.getElementById('regError');

      errorEl.classList.add('hidden');
      errorEl.textContent = '';

      if (!name || !email || !pass || pass.length < 6) {
        errorEl.textContent = "Fill all fields. Password min 6 chars.";
        errorEl.classList.remove('hidden');
        return;
      }

      if (registeredUsers.some(u => u.email.toLowerCase() === email)) {
        errorEl.textContent = "Email already taken.";
        errorEl.classList.remove('hidden');
        return;
      }

      registeredUsers.push({ email, pass, name });
      saveAllData();
      hideRegisterModal();
      alert(`Registered! You can now log in with ${email}.`);
    }

    function handleLogout() {
      if (confirm("Log out now?")) {
        localStorage.removeItem('currentUser');
        currentUser = null;
        isAdmin = false;
        lockSite();

    function addOfficer() {
      const name = document.getElementById('newOfficerName').value.trim();
      const role = document.getElementById('newOfficerRole').value.trim();
      const phone = document.getElementById('newOfficerPhone').value.trim() || 'N/A';
      const email = document.getElementById('newOfficerEmail').value.trim() || 'N/A';

      if (!name || !role) return alert("Name and Role required");

      const newId = officerContacts.length ? Math.max(...officerContacts.map(c => c.id)) + 1 : 1;
      officerContacts.push({ id: newId, name, role, phone, email });
      saveAllData();
      renderOfficersAdmin();
      renderClientOfficers();
      alert("Officer added!");
    }

    function addMatchRecord() {
      const date = document.getElementById('matchDate').value;
      const team1 = document.getElementById('matchTeam1').value;
      const team2 = document.getElementById('matchTeam2').value;
      const score = document.getElementById('matchScore').value;
      const winner = document.getElementById('matchWinner').value;
      const file = document.getElementById('matchImage').files[0];

      if (!date || !team1 || !team2) return alert("Fill required fields");

      const reader = new FileReader();
      reader.onload = function(e) {
        matchRecords.push({ id: Date.now(), date, team1, team2, score, winner, image: e.target.result });
        saveAllData();
        renderMatchAdmin();
        renderClientMatches();
        alert("Match added with image!");
      };
      if (file) reader.readAsDataURL(file);
      else {
        matchRecords.push({ id: Date.now(), date, team1, team2, score, winner, image: null });
        saveAllData();
        renderMatchAdmin();
        renderClientMatches();
        alert("Match added!");
      }
    }
